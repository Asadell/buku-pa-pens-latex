% Toggle Front Matter Opsional
% Set ke true untuk menampilkan seksi yang diinginkan.

\newif\ifshowabstract      \showabstractfalse      % Abstract (EN)

\newif\ifshowdaftargambar  \showdaftargambarfalse  % Daftar Gambar

% Toggle Tipe Dokumen

% \proposalfalse -> kompilasi sebagai Progres PA / Buku PA (Bab 1-5)

% Toggle Paksa Halaman Ganjil
% Jika true, setiap bab dan seksi front matter selalu dimulai
% di halaman ganjil (sisipkan halaman kosong jika perlu).
% Buku PA: aktif otomatis (openright).
% Proposal & Progres: default false (oneside, tidak memerlukan ini).
% Aktifkan dengan menambahkan \showforceoddpagetrue di main_*.tex.

% Toggle Jenis Dokumen: Buku PA
